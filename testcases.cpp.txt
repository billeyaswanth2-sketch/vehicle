#include <chrono>
#include <iostream>
#include <map>
#include <string>

// ---------------------------------------------------------------------------
// Constants
// ---------------------------------------------------------------------------

constexpr int WAIT_TIMEOUT_MS = 500;

// ---------------------------------------------------------------------------
// Minimal SysVar simulator (replaces CANoe environment variables)
// ---------------------------------------------------------------------------

class SysVarSimulator {
public:
    SysVarSimulator() {
        vars_["ignition"] = 0;
        vars_["brake"]    = 1;  // pre-set to 1 (brake pressed by default)
        vars_["gear"]     = 0;
        vars_["engine"]   = 0;
    }

    void set(const std::string& name, int value) {
        vars_[name] = value;
    }

    int get(const std::string& name) const {
        auto it = vars_.find(name);
        if (it != vars_.end()) return it->second;
        return -1;
    }

    /**
     * Returns 1 if the variable is accessible (always in simulation).
     * In a real CANoe environment this would block until the variable
     * changes or the timeout expires, returning 0 on timeout.
     */
    int waitForSysVar(const std::string& /*name*/, int timeout_ms) {
        // tiny sim delay (1% of real timeout)
        std::this_thread::sleep_for(
            std::chrono::milliseconds(timeout_ms / 100));
        return 1;  // 1 = received, 0 = timeout
    }

private:
    std::map<std::string, int> vars_;
};

// ---------------------------------------------------------------------------
// Test report helper (mirrors CAPL testStepPass / testStepFail)
// ---------------------------------------------------------------------------

struct StepResult {
    std::string step;
    std::string status;  // "PASS" or "FAIL"
    std::string msg;
};

class TestReport {
public:
    TestReport(const std::string& module_title,
               const std::string& description,
               const std::string& engineer)
        : module_title_(module_title),
          description_(description),
          engineer_(engineer) {}

    void stepPass(const std::string& step, const std::string& message) {
        results_.push_back({step, "PASS", message});
        std::cout << "  [PASS] " << step << ": " << message << "\n";
    }

    void stepFail(const std::string& step, const std::string& message) {
        results_.push_back({step, "FAIL", message});
        std::cout << "  [FAIL] " << step << ": " << message << "\n";
    }

    void summary() const {
        int passed = 0, failed = 0;
        for (const auto& r : results_) {
            if (r.status == "PASS") ++passed;
            else                    ++failed;
        }
        std::cout << "\n" << std::string(55, '=') << "\n";
        std::cout << "Module : " << module_title_ << "\n";
        std::cout << "Eng    : " << engineer_     << "\n";
        std::cout << "Total  : " << results_.size()
                  << "  |  Pass: " << passed
                  << "  |  Fail: " << failed << "\n";
        std::cout << std::string(55, '=') << "\n";
    }

    const std::vector<StepResult>& results() const { return results_; }

private:
    std::string              module_title_;
    std::string              description_;
    std::string              engineer_;
    std::vector<StepResult>  results_;
};

// ---------------------------------------------------------------------------
// Shared test environment (module-scoped fixture)
// ---------------------------------------------------------------------------

class VpdEnvironment : public ::testing::Environment {
public:
    static SysVarSimulator* sv;
    static TestReport*      report;

    void SetUp() override {
        sv     = new SysVarSimulator();
        report = new TestReport(
            "vehicle power distribution",
            "test cases are written in C++ (converted from CAPL)",
            "Yaswanth");

        std::cout << std::string(55, '=') << "\n";
        std::cout << "Module : vehicle power distribution\n";
        std::cout << "Desc   : test cases are written in C++ (converted from CAPL)\n";
        std::cout << "Eng    : Yaswanth\n";
        std::cout << std::string(55, '=') << "\n";

        std::this_thread::sleep_for(std::chrono::milliseconds(400));
        std::cout << "\n--- Group: simulation of signals / test mode communication ---\n";
    }

    void TearDown() override {
        report->summary();
        delete sv;
        delete report;
    }
};

SysVarSimulator* VpdEnvironment::sv     = nullptr;
TestReport*      VpdEnvironment::report = nullptr;

// ---------------------------------------------------------------------------
// Helper: count FAIL results added after 'before_size'
// ---------------------------------------------------------------------------

static std::vector<StepResult> newFailures(const TestReport& report,
                                           size_t before_size) {
    std::vector<StepResult> fails;
    const auto& all = report.results();
    for (size_t i = before_size; i < all.size(); ++i) {
        if (all[i].status == "FAIL") fails.push_back(all[i]);
    }
    return fails;
}

// ---------------------------------------------------------------------------
// tc_brake
// Verifies that setting the brake sysvar to 1 is acknowledged correctly.
// ---------------------------------------------------------------------------

static void tc_brake(SysVarSimulator& sv, TestReport& report) {
    std::cout << "\n[TC] brake_test\n";

    sv.set("brake", 1);
    int wait_result = sv.waitForSysVar("brake", WAIT_TIMEOUT_MS);

    if (wait_result == 0) {
        report.stepFail("test step1", "sys var is not received");
    } else {
        if (sv.get("brake") == 1)
            report.stepPass("test step1", "brake is working fine");
        else
            report.stepFail("test step1", "brake is not working");
    }

    std::this_thread::sleep_for(std::chrono::milliseconds(WAIT_TIMEOUT_MS));
}

// ---------------------------------------------------------------------------
// tc_gear
// Validates gear positions 0-5 (Park, Reverse, Neutral, Drive, 3rd, 2nd).
// ---------------------------------------------------------------------------

static void tc_gear(SysVarSimulator& sv, TestReport& report) {
    std::cout << "\n[TC] gear_test\n";

    struct GearStep {
        int         value;
        std::string pass_msg;
    };

    const std::vector<GearStep> gear_steps = {
        {0, "gear is in Park / Neutral"},
        {1, "gear 1 is activated"},
        {2, "gear 2 is activated"},
        {3, "gear 3 is activated"},
        {4, "gear 4 is activated"},
        {5, "gear 5 is activated"},
    };

    int step_num = 1;
    for (const auto& gs : gear_steps) {
        std::string step_label = "test step" + std::to_string(step_num++);
        sv.set("gear", gs.value);
        int wait_result = sv.waitForSysVar("gear", WAIT_TIMEOUT_MS);

        if (wait_result == 0) {
            report.stepFail(step_label, "sys var is not received");
        } else {
            if (sv.get("gear") == gs.value)
                report.stepPass(step_label, gs.pass_msg);
            else
                report.stepFail(step_label, "gear is not working");
        }

        std::this_thread::sleep_for(std::chrono::milliseconds(WAIT_TIMEOUT_MS));
    }
}

// ---------------------------------------------------------------------------
// tc_ignition_key
// Tests all four ignition key positions:
//   0 = Lock, 1 = Accessory, 2 = ON, 3 = Start
// For position 3 (Start) the engine sysvar is set to 1 on pass,
// and gear must be 0 + brake must be 1 (safety interlock).
// ---------------------------------------------------------------------------

static void tc_ignition_key(SysVarSimulator& sv, TestReport& report) {
    std::cout << "\n[TC] ignition_test\n";

    struct IgnStep {
        int         value;
        std::string pass_msg;
        std::string fail_msg;
    };

    const std::vector<IgnStep> ignition_steps = {
        {0, "key is in lock mode",      "key is not working"},
        {1, "key is in accessory mode", "key is not working"},
        {2, "key is in ON mode",        "key is not working"},
    };

    int step_num = 1;
    for (const auto& is : ignition_steps) {
        std::string step_label = "test step" + std::to_string(step_num++);
        sv.set("ignition", is.value);
        int wait_result = sv.waitForSysVar("ignition", WAIT_TIMEOUT_MS);

        if (wait_result == 0) {
            report.stepFail(step_label, "sys var is not received");
        } else {
            if (sv.get("ignition") == is.value)
                report.stepPass(step_label, is.pass_msg);
            else
                report.stepFail(step_label, is.fail_msg);
        }

        std::this_thread::sleep_for(std::chrono::milliseconds(WAIT_TIMEOUT_MS));
    }

    // Step 4 - Start (ignition = 3), requires brake=1 and gear=0
    sv.set("gear",     0);
    sv.set("ignition", 3);
    int wait_result = sv.waitForSysVar("ignition", WAIT_TIMEOUT_MS);

    if (wait_result == 0) {
        report.stepFail("test step4", "sys var is not received");
    } else {
        bool ign_ok   = sv.get("ignition") == 3;
        bool brake_ok = sv.get("brake")    == 1;
        bool gear_ok  = sv.get("gear")     == 0;

        if (ign_ok && brake_ok && gear_ok) {
            report.stepPass("test step4", "key is in start mode");
            sv.set("engine", 1);  // engine cranks
        } else {
            report.stepFail("test step4", "key is not working");
        }
    }

    std::this_thread::sleep_for(std::chrono::milliseconds(WAIT_TIMEOUT_MS));
}

// ---------------------------------------------------------------------------
// GTest test cases
// ---------------------------------------------------------------------------

TEST(VpdTest, BrakeTest) {
    auto* sv     = VpdEnvironment::sv;
    auto* report = VpdEnvironment::report;
    ASSERT_NE(sv,     nullptr);
    ASSERT_NE(report, nullptr);

    size_t before = report->results().size();
    tc_brake(*sv, *report);

    auto fails = newFailures(*report, before);
    EXPECT_TRUE(fails.empty())
        << "Brake test had " << fails.size() << " failure(s). "
        << "First: " << (fails.empty() ? "" : fails[0].msg);
}
